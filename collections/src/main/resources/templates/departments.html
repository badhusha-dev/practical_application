// Map initialization
Map<Integer, Department> departments = new HashMap<>();

// Adding departments
departments.put(1, new Department("IT"));
departments.put(2, new Department("HR"));

// Fast lookup by ID
Department dept = departments.get(1); // O(1) access

// Iteration
departments.forEach((id, dept) -> {
    System.out.println("Department " + id + ": " + dept.getName());
});
